"""Data models for the task manager."""

from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from enum import Enum
from typing import Optional


class TaskStatus(str, Enum):
    """Possible states for a task."""
    PENDING = "pending"
    COMPLETED = "completed"


@dataclass(frozen=True)
class Task:
    """Immutable representation of a task row."""
    id: int
    title: str
    description: Optional[str]
    status: TaskStatus
    created_at: datetime
    completed_at: Optional[datetime] = None

    @classmethod
    def from_row(cls, row: tuple) -> Task:
        """
        Construct a Task from a raw SQLite row.
        Expected column order: id, title, description, status, created_at, completed_at
        """
        id_, title, description, status_str, created_at_str, completed_at_str = row
        return cls(
            id=id_,
            title=title,
            description=description,
            status=TaskStatus(status_str),
            created_at=datetime.fromisoformat(created_at_str),
            completed_at=(
                datetime.fromisoformat(completed_at_str)
                if completed_at_str
                else None
            ),
        )

    @property
    def is_completed(self) -> bool:
        return self.status == TaskStatus.COMPLETED

"""SQLite database layer — all SQL lives here."""

from __future__ import annotations

import sqlite3
from contextlib import contextmanager
from datetime import datetime
from pathlib import Path
from typing import Generator, List, Optional

from task_manager.config import get_db_path
from task_manager.models import Task, TaskStatus


# ── Schema ───────────────────────────────────────────────────────────────────

_SCHEMA_SQL = """\
CREATE TABLE IF NOT EXISTS tasks (
    id           INTEGER PRIMARY KEY AUTOINCREMENT,
    title        TEXT    NOT NULL,
    description  TEXT,
    status       TEXT    NOT NULL DEFAULT 'pending'
                         CHECK (status IN ('pending', 'completed')),
    created_at   TEXT    NOT NULL,
    completed_at TEXT
);
"""


class TaskDatabase:
    """Thin wrapper around an SQLite connection scoped to one DB file."""

    def __init__(self, db_path: Optional[Path] = None) -> None:
        self.db_path = db_path or get_db_path()
        self._connection: Optional[sqlite3.Connection] = None
        self._ensure_schema()

    # ── Connection management ────────────────────────────────────────────

    @property
    def connection(self) -> sqlite3.Connection:
        if self._connection is None:
            self._connection = sqlite3.connect(str(self.db_path))
            # Enable WAL mode for better concurrent read performance
            self._connection.execute("PRAGMA journal_mode=WAL;")
            # Enforce FK constraints
            self._connection.execute("PRAGMA foreign_keys=ON;")
        return self._connection

    def close(self) -> None:
        if self._connection is not None:
            self._connection.close()
            self._connection = None

    @contextmanager
    def _transaction(self) -> Generator[sqlite3.Cursor, None, None]:
        """Context manager that commits on success, rolls back on error."""
        cursor = self.connection.cursor()
        try:
            yield cursor
            self.connection.commit()
        except Exception:
            self.connection.rollback()
            raise

    def _ensure_schema(self) -> None:
        with self._transaction() as cur:
            cur.executescript(_SCHEMA_SQL)

    # ── CRUD operations ──────────────────────────────────────────────────

    def add_task(self, title: str, description: Optional[str] = None) -> Task:
        """Insert a new task and return the created Task object."""
        now = datetime.now().isoformat(timespec="seconds")
        with self._transaction() as cur:
            cur.execute(
                "INSERT INTO tasks (title, description, status, created_at) "
                "VALUES (?, ?, ?, ?)",
                (title, description, TaskStatus.PENDING.value, now),
            )
            task_id = cur.lastrowid

        return self.get_task(task_id)  # type: ignore[arg-type]

    def get_task(self, task_id: int) -> Optional[Task]:
        """Fetch a single task by ID, or None if not found."""
        cur = self.connection.execute(
            "SELECT id, title, description, status, created_at, completed_at "
            "FROM tasks WHERE id = ?",
            (task_id,),
        )
        row = cur.fetchone()
        return Task.from_row(row) if row else None

    def list_tasks(self, status_filter: Optional[TaskStatus] = None) -> List[Task]:
        """
        Return tasks ordered by creation date (newest first).
        Optionally filter by status.
        """
        if status_filter is not None:
            cur = self.connection.execute(
                "SELECT id, title, description, status, created_at, completed_at "
                "FROM tasks WHERE status = ? ORDER BY created_at DESC",
                (status_filter.value,),
            )
        else:
            cur = self.connection.execute(
                "SELECT id, title, description, status, created_at, completed_at "
                "FROM tasks ORDER BY created_at DESC",
            )
        return [Task.from_row(row) for row in cur.fetchall()]

    def complete_task(self, task_id: int) -> Task:
        """
        Mark a task as completed. Raises ValueError if not found
        or already completed.
        """
        task = self.get_task(task_id)
        if task is None:
            raise ValueError(f"Task with ID {task_id} not found.")
        if task.is_completed:
            raise ValueError(f"Task {task_id} is already completed.")

        now = datetime.now().isoformat(timespec="seconds")
        with self._transaction() as cur:
            cur.execute(
                "UPDATE tasks SET status = ?, completed_at = ? WHERE id = ?",
                (TaskStatus.COMPLETED.value, now, task_id),
            )

        return self.get_task(task_id)  # type: ignore[return-value]

    def delete_task(self, task_id: int) -> Task:
        """
        Delete a task. Returns the deleted task. Raises ValueError if not found.
        """
        task = self.get_task(task_id)
        if task is None:
            raise ValueError(f"Task with ID {task_id} not found.")

        with self._transaction() as cur:
            cur.execute("DELETE FROM tasks WHERE id = ?", (task_id,))

        return task

    def count_tasks(self) -> dict[str, int]:
        """Return a dict with total / pending / completed counts."""
        cur = self.connection.execute(
            "SELECT status, COUNT(*) FROM tasks GROUP BY status"
        )
        counts: dict[str, int] = {"total": 0, "pending": 0, "completed": 0}
        for status_val, cnt in cur.fetchall():
            counts[status_val] = cnt
            counts["total"] += cnt
        return counts

from rich.table import Table
from rich.text import Text

from task_manager.db import TaskDatabase
from task_manager.models import Task, TaskStatus

# ── Typer app ────────────────────────────────────────────────────────────────

app = typer.Typer(
    name="tasks",
    help="✅ A simple CLI task manager with SQLite storage.",
    add_completion=False,
    rich_markup_mode="rich",
    no_args_is_help=True,
)

console = Console()
err_console = Console(stderr=True)


# ── Helpers ──────────────────────────────────────────────────────────────────

def _get_db() -> TaskDatabase:
    """Instantiate the database (schema auto-created on first use)."""
    return TaskDatabase()


def _status_badge(task: Task) -> Text:
    """Return a coloured status indicator."""
    if task.is_completed:
        return Text("✔ Done", style="bold green")
    return Text("● Pending", style="bold yellow")


def _render_task_row(table: Table, task: Task) -> None:
    """Add a single task as a row in a Rich table."""
    title_style = "dim strike" if task.is_completed else "bold"
    desc = task.description or "—"
    if len(desc) > 60:
        desc = desc[:57] + "..."

    table.add_row(
        str(task.id),
        Text(task.title, style=title_style),
        desc,
        _status_badge(task),
        task.created_at.strftime("%Y-%m-%d %H:%M"),
        (
            task.completed_at.strftime("%Y-%m-%d %H:%M")
            if task.completed_at
            else "—"
        ),
    )


def _build_task_table(tasks: list[Task], title: str) -> Table:
    """Construct a Rich table pre-populated with task rows."""
    table = Table(
        title=title,
        box=box.ROUNDED,
        show_lines=False,
        header_style="bold cyan",
        title_style="bold magenta",
        padding=(0, 1),
    )
    table.add_column("ID", justify="right", style="dim", width=5)
    table.add_column("Title", min_width=20, max_width=40)
    table.add_column("Description", min_width=15, max_width=40)
    table.add_column("Status", width=12)
    table.add_column("Created", width=18)
    table.add_column("Completed", width=18)

    for task in tasks:
        _render_task_row(table, task)

    return table

    if not title.strip():
