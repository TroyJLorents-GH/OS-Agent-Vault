-- SQLite database schema for task_cli

PRAGMA journal_mode = WAL;          -- Better concurrent read performance
PRAGMA foreign_keys = ON;           -- Enforce FK constraints

CREATE TABLE IF NOT EXISTS tasks (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    title       TEXT    NOT NULL CHECK(length(title) > 0 AND length(title) <= 200),
    description TEXT    DEFAULT '' CHECK(length(description) <= 4000),
    status      TEXT    NOT NULL DEFAULT 'pending'
                        CHECK(status IN ('pending', 'in_progress', 'done', 'archived')),
    priority    INTEGER NOT NULL DEFAULT 0 CHECK(priority BETWEEN 0 AND 3),
                        -- 0=none, 1=low, 2=medium, 3=high
    created_at  TEXT    NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now')),
    updated_at  TEXT    NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
);

-- Index for the most common query: list tasks filtered/sorted by status
CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);

-- Index for sorting by creation date (default list order)
CREATE INDEX IF NOT EXISTS idx_tasks_created_at ON tasks(created_at DESC);

-- Schema versioning table for future migrations
CREATE TABLE IF NOT EXISTS schema_migrations (
    version     INTEGER PRIMARY KEY,
    applied_at  TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
);

-- Trigger: auto-update `updated_at` on any row modification
CREATE TRIGGER IF NOT EXISTS trg_tasks_updated_at
AFTER UPDATE ON tasks
FOR EACH ROW
WHEN OLD.updated_at = NEW.updated_at   -- Only fire if caller didn't set it explicitly
BEGIN
    UPDATE tasks SET updated_at = strftime('%Y-%m-%dT%H:%M:%SZ', 'now')
    WHERE id = NEW.id;
END;

"""Domain models — pure data, no framework dependencies."""

from __future__ import annotations

import enum
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Optional


class Status(str, enum.Enum):
    """Allowed task statuses. Values match the DB CHECK constraint."""
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    DONE = "done"
    ARCHIVED = "archived"


class Priority(int, enum.Enum):
    NONE = 0
    LOW = 1
    MEDIUM = 2
    HIGH = 3


@dataclass
class Task:
    """Core domain object. Maps 1-to-1 with the `tasks` table."""
    id: Optional[int]               # None before insertion
    title: str
    description: str = ""
    status: Status = Status.PENDING
    priority: Priority = Priority.NONE
    created_at: datetime = field(default_factory=lambda: datetime.now(timezone.utc))
    updated_at: datetime = field(default_factory=lambda: datetime.now(timezone.utc))

    # --- Convenience helpers ---

    @property
    def is_complete(self) -> bool:
        return self.status == Status.DONE

    def validate(self) -> None:
        """Raise ValueError if business rules are violated."""
        if not self.title or not self.title.strip():
            raise ValueError("Task title must not be empty.")
        if len(self.title) > 200:
            raise ValueError("Task title must be 200 characters or fewer.")
        if len(self.description) > 4000:
            raise ValueError("Task description must be 4000 characters or fewer.")

    @classmethod
    def from_row(cls, row: dict) -> Task:
        """Construct a Task from a sqlite3.Row dict."""
        return cls(
            id=row["id"],
            title=row["title"],
            description=row["description"],
            status=Status(row["status"]),
            priority=Priority(row["priority"]),
            created_at=datetime.fromisoformat(row["created_at"]),
            updated_at=datetime.fromisoformat(row["updated_at"]),
        )

# task_cli/database.py
"""SQLite connection management and schema bootstrapping."""

from __future__ import annotations

import sqlite3
from contextlib import contextmanager
from pathlib import Path
from typing import Generator

# Schema lives alongside this file; could also be inlined.
_SCHEMA_SQL = (Path(__file__).parent / ".." / "database_schema.sql").resolve()

# Default DB location: ~/.task_cli/tasks.db
DEFAULT_DB_PATH = Path.home() / ".task_cli" / "tasks.db"

CURRENT_SCHEMA_VERSION = 1


def get_connection(db_path: Path = DEFAULT_DB_PATH) -> sqlite3.Connection:
    """Return a configured connection. Creates parent dirs if needed."""
    db_path.parent.mkdir(parents=True, exist_ok=True)
    conn = sqlite3.connect(str(db_path))
    conn.row_factory = sqlite3.Row          # Access columns by name
    conn.execute("PRAGMA journal_mode = WAL")
    conn.execute("PRAGMA foreign_keys = ON")
    return conn


@contextmanager
def transaction(conn: sqlite3.Connection) -> Generator[sqlite3.Cursor, None, None]:
    """Context manager that commits on success, rolls back on exception."""
    cur = conn.cursor()
    try:
        yield cur
        conn.commit()
    except Exception:
        conn.rollback()
        raise
    finally:
        cur.close()


def initialize(conn: sqlite3.Connection) -> None:
    """Run schema creation (idempotent via IF NOT EXISTS)."""
    schema_sql = """
    CREATE TABLE IF NOT EXISTS tasks (
        id          INTEGER PRIMARY KEY AUTOINCREMENT,
        title       TEXT    NOT NULL CHECK(length(title) > 0 AND length(title) <= 200),
        description TEXT    DEFAULT '' CHECK(length(description) <= 4000),
        status      TEXT    NOT NULL DEFAULT 'pending'
                            CHECK(status IN ('pending', 'in_progress', 'done', 'archived')),
        priority    INTEGER NOT NULL DEFAULT 0 CHECK(priority BETWEEN 0 AND 3),
        created_at  TEXT    NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now')),
        updated_at  TEXT    NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
    );
    CREATE INDEX IF NOT EXISTS idx_tasks_status ON tasks(status);
    CREATE INDEX IF NOT EXISTS idx_tasks_created_at ON tasks(created_at DESC);

    CREATE TABLE IF NOT EXISTS schema_migrations (
        version    INTEGER PRIMARY KEY,
        applied_at TEXT NOT NULL DEFAULT (strftime('%Y-%m-%dT%H:%M:%SZ', 'now'))
    );

    CREATE TRIGGER IF NOT EXISTS trg_tasks_updated_at
    AFTER UPDATE ON tasks
    FOR EACH ROW
    WHEN OLD.updated_at = NEW.updated_at
    BEGIN
        UPDATE tasks SET updated_at = strftime('%Y-%m-%dT%H:%M:%SZ', 'now')
        WHERE id = NEW.id;
    END;
    """
    with transaction(conn) as cur:
        cur.executescript(schema_sql)
        # Record migration version (idempotent)
        cur.execute(
            "INSERT OR IGNORE INTO schema_migrations (version) VALUES (?)",
            (CURRENT_SCHEMA_VERSION,),
        )

# task_cli/repository.py
"""CRUD operations against the tasks table."""

from __future__ import annotations

import sqlite3
from typing import List, Optional

from .database import transaction
from .models import Priority, Status, Task


class TaskRepository:
    """Thin data-access layer. Receives a connection; owns no connection lifecycle."""

    def __init__(self, conn: sqlite3.Connection) -> None:
        self._conn = conn

    # ---- CREATE ----

    def add(self, title: str, description: str = "", priority: Priority = Priority.NONE) -> Task:
        """Insert a new task and return it with its generated id."""
        sql = """
            INSERT INTO tasks (title, description, priority)
            VALUES (?, ?, ?)
        """
        with transaction(self._conn) as cur:
            cur.execute(sql, (title, description, priority.value))
            task_id = cur.lastrowid
        return self.get(task_id)  # type: ignore[arg-type]

    # ---- READ ----

    def get(self, task_id: int) -> Optional[Task]:
        """Fetch a single task by primary key."""
        cur = self._conn.execute("SELECT * FROM tasks WHERE id = ?", (task_id,))
        row = cur.fetchone()
        return Task.from_row(row) if row else None

    def list(
        self,
        status: Optional[Status] = None,
        sort_by: str = "created_at",
        limit: int = 50,
    ) -> List[Task]:
        """Return tasks with optional status filter, sorted descending."""
        allowed_sort = {"created_at", "updated_at", "priority", "title", "status"}
        if sort_by not in allowed_sort:
            raise ValueError(f"sort_by must be one of {allowed_sort}")

        clauses: list[str] = []
        params: list = []

        if status is not None:
            clauses.append("status = ?")
            params.append(status.value)

        where = f"WHERE {' AND '.join(clauses)}" if clauses else ""
        sql = f"SELECT * FROM tasks {where} ORDER BY {sort_by} DESC LIMIT ?"
        params.append(limit)

        cur = self._conn.execute(sql, params)
        return [Task.from_row(row) for row in cur.fetchall()]

    # ---- UPDATE ----

    def complete(self, task_id: int) -> Optional[Task]:
        """Mark a task as done. Returns updated task or None if not found."""
        sql = "UPDATE tasks SET status = ? WHERE id = ?"
        with transaction(self._conn) as cur:
            cur.execute(sql, (Status.DONE.value, task_id))
            if cur.rowcount == 0:
                return None
        return self.get(task_id)

    def edit(
        self,
        task_id: int,
        title: Optional[str] = None,
        description: Optional[str] = None,
        status: Optional[Status] = None,
        priority: Optional[Priority] = None,
    ) -> Optional[Task]:
        """Partial update — only supplied fields are changed."""
        fields: list[str] = []
        params: list = []
        if title is not None:
            fields.append("title = ?")
            params.append(title)
        if description is not None:
            fields.append("description = ?")
            params.append(description)
        if status is not None:
            fields.append("status = ?")
            params.append(status.value)
        if priority is not None:
            fields.append("priority = ?")
            params.append(priority.value)

        if not fields:
            return self.get(task_id)

        params.append(task_id)
        sql = f"UPDATE tasks SET {', '.join(fields)} WHERE id = ?"
        with transaction(self._conn) as cur:
            cur.execute(sql, params)
            if cur.rowcount == 0:
                return None
        return self.get(task_id)

    # ---- DELETE ----

    def delete(self, task_id: int) -> bool:
        """Remove a task. Returns True if a row was actually deleted."""
        sql = "DELETE FROM tasks WHERE id = ?"
        with transaction(self._conn) as cur:
            cur.execute(sql, (task_id,))
            return cur.rowcount > 0

# task_cli/cli.py
"""Click-based CLI command definitions."""

from __future__ import annotations

import sys

import click

from .config import resolve_db_path
from .database import get_connection, initialize
from .formatting import format_task_detail, format_task_table
from .models import Priority, Status
from .repository import TaskRepository


# ── Shared plumbing ──────────────────────────────────────────────

@click.group()
@click.option(
    "--db",
    envvar="TASK_CLI_DB",
    default=None,
    type=click.Path(),
    help="Path to SQLite database file [default: ~/.task_cli/tasks.db]",
)
@click.pass_context
def cli(ctx: click.Context, db: str | None) -> None:
    """task — A minimal CLI task manager backed by SQLite."""
    ctx.ensure_object(dict)
    db_path = resolve_db_path(db)
    conn = get_connection(db_path)
    initialize(conn)
    ctx.obj["repo"] = TaskRepository(conn)
    ctx.obj["conn"] = conn


# ── ADD ──────────────────────────────────────────────────────────

@cli.command()
@click.option("-t", "--title", required=True, help="Task title (max 200 chars).")
@click.option("-d", "--desc", default="", help="Optional description.")
@click.option(
    "-p", "--priority",
    type=click.Choice(["none", "low", "medium", "high"], case_sensitive=False),
    default="none",
    help="Priority level.",
)
@click.pass_context
def add(ctx: click.Context, title: str, desc: str, priority: str) -> None:
    """Create a new task."""
    repo: TaskRepository = ctx.obj["repo"]
    pri = Priority[priority.upper()]
    task = repo.add(title=title, description=desc, priority=pri)
    click.echo(f"✓ Created task #{task.id}")
